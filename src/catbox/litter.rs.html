3//! Functions for handling temporary file upload through Litterbox's API
4//!
5//! Calls API described at <https://litterbox.catbox.moe/tools.php>.
6//!
7//! See <https://litterbox.catbox.moe/faq.php> for allowed filetypes and content.
8
9use anyhow::Result;
10use reqwest::{ Client, multipart::{ Form, Part } };
11
12use crate::{ LITTER_API_URL, UASTRING, helper::{ file_name, file_stream } };
13
14/// Upload a temporary file to litterbox.
15/// Max size 1GB.
16///
17/// See <https://litterbox.catbox.moe/faq.php> for allowed formats and content.
18///
19/// # Arguments
20///
21/// * `file_path` - Path to the file to be uploaded
22/// * `time` - Time until expiration. Valid values are 1, 12, 24 and 72.
23pub async fn upload<S: Into<String>>(file_path: S, time: u8) -> Result<String> {
24    let file_path = file_path.into();
25    let file = file_stream(&file_path).await?;
26    let file_name = file_name(&file_path);
27
28    let form = Form::new()
29        .text("reqtype", "fileupload")
30        .text("time", format!("{time}h"))
31        .part("fileToUpload", Part::stream(file).file_name(file_name));
32
33    Ok(
34        Client::builder()
35            .user_agent(UASTRING)
36            .build()
37            .unwrap_or_else(|_| Client::new())
38            .post(LITTER_API_URL)
39            .multipart(form)
40            .send().await?
41            .text().await?
42    )
43}
