3//! Functions for handling file upload and deletion through Catbox's API

15/// Upload a file to catbox.
16///
17/// Returns an URL to the file
18///
19/// See <https://catbox.moe/faq.php> for allowed formats and content.
20///
21/// # Arguments
22///
23/// * `file_path` - Path to the file to be uploaded
24/// * `user_hash` - User account hash, required for deletions. Otherwise optional.
25pub async fn from_file<S: Into<String>>(file_path: S, user_hash: S) -> Result<String> {
26    let file_path = file_path.into();
27    let file = file_stream(&file_path).await?;
28    let file_name = file_name(&file_path);
29
30    let form = Form::new()
31        .text("reqtype", "fileupload")
32        .text("userhash", user_hash.into())
33        .part("fileToUpload", Part::stream(file).file_name(file_name));
34
35    Ok(
36        Client::builder()
37            .user_agent(UASTRING)
38            .build()
39            .unwrap_or_else(|_| Client::new())
40            .post(CATBOX_API_URL)
41            .multipart(form)
42            .send().await?
43            .text().await?
44    )
45}

47/// Upload contents from a URL to catbox
48///
49/// Returns n URL to the file
50///
51/// See <https://catbox.moe/faq.php> for allowed formats and content.
52///
53/// # Arguments
54///
55/// * `url` - URL to file
56/// * `user_hash` - User account hash, required for deletions. Otherwise optional.
57pub async fn from_url<S: Into<String>>(url: S, user_hash: S) -> Result<String> {
58    let form = [
59        ("reqtype", "urlupload"),
60        ("userhash", &user_hash.into()),
61        ("url", &url.into()),
62    ];
63
64    Ok(
65        Client::builder()
66            .user_agent(UASTRING)
67            .build()
68            .unwrap_or_else(|_| Client::new())
69            .post(CATBOX_API_URL)
70            .form(&form)
71            .send().await?
72            .text().await?
73    )
74}

76/// Delete files
77///
78/// Returns "Files successfully deleted." on success
79///
80/// # Arguments
81///
82/// * `files` - Names of the files to be deleted
83/// * `user_hash` - User account hash, required for deletions. Otherwise optional.
84pub async fn delete<S: Into<String>>(files: Vec<S>, user_hash: S) -> Result<String> {
85    let files: Vec<_> = files.into_iter().map(Into::into).collect();
86
87    let form = [
88        ("reqtype", "deletefiles"),
89        ("userhash", &user_hash.into()),
90        ("files", &files.join(" ")),
91    ];
92
93    Ok(
94        Client::builder()
95            .user_agent(UASTRING)
96            .build()
97            .unwrap_or_else(|_| Client::new())
98            .post(CATBOX_API_URL)
99            .form(&form)
100            .send().await?
101            .text().await?
102    )
103}
